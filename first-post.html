def movement(num):
    if num == 5:
        num = random.choice([2,4,6,8])
    elif num == 1:
        num = random.choice([2,4])
    elif num == 2:
        num = random.choice([1,3,5])
    elif num == 3:
        num = random.choice([2,6])
    elif num == 4:
        num = random.choice([1,5,7])
    elif num == 6:
        num = random.choice([3,5,9])
    elif num == 7:
        num = random.choice([4,8])
    elif num == 8:
        num = random.choice([5,7,9])
    elif num == 9:
        num = random.choice([6,8])
    return num

def simulation():
    num = 5
    for i in range(500):
        num = movement(num)
    return num

def gamesWon(bet, win_rate):
    result = 0
    bet = round(bet)
    bets = np.random.rand(bet)
    for rand in bets:
        if rand >= 1 - win_rate:
            result += 1
    return result
