Predicting Gas Pollutant Concentrations in the US¶

The purpose of this capstone was to gather air pollution data from WHO, analyze carbon monoxide(CO), sulfur dioxide(SO2), nitrogen dioxide(NO2), and bad ozone(O3) concentrations from 2000-2016 and predict these concentrations for the future.

Analytical Questions:
1.Is air pollution getting better or worse over time in the US?
2.How well does ARIMA and VARMA capture the data?

Hypothesis
I believe that the data will show a general increase in concentration of gases as time went on from 2000-2016. Also, I believe that my predictions from the modeling will show the same general trend.

def expandgrid(*itrs):
    product = list(itertools.product(*itrs))
    return {'Var{}'.format(i+1):[x[i] for x in product] for i in range(len(itrs))}

a = [x for x in range(2000,2017)]
b = [x for x in range(1,13)]

month_year = pd.DataFrame(expandgrid(a,b))
month_year = month_year.rename(columns={"Var1": "year", "Var2": "month"})

def getGasMonthlyAvg(gas, month_year):
    temp = df.groupby(["year", "month"]).mean().reset_index()
    temp2 = month_year.merge(temp, on=["year", "month"], how="left")
    temp2 = temp2.fillna(method="ffill")
    temp2 = temp2[gas + " Mean"]
    return temp2

#Gets the averages of each gas per year puts it in json format with state and year
df_years = df.groupby("year")
gas_avg = []
for year in years:
    df_current_year = df_years.get_group(year)
    gas_year = []
    for state in states:
        if state in df_current_year.state.unique():
            dic = {}
            df_states = df_current_year.groupby("state")
            df_current_state = df_states.get_group(state)
            NO2_mean = df_current_state["NO2 Mean"].mean()
            O3_mean = df_current_state["O3 Mean"].mean()
            SO2_mean = df_current_state["SO2 Mean"].mean()
            CO_mean = df_current_state["CO Mean"].mean()
            dic[str(year) + "_" + "NO2_mean"] = NO2_mean
            dic[str(year) + "_" + "O3_mean"] = O3_mean
            dic[str(year) + "_" + "SO2_mean"] = SO2_mean
            dic[str(year) + "_" + "CO_mean"] = CO_mean
            gas_year.append(dic)
        else:
            dic = {}
            dic[str(year) + "_" + "NO2_mean"] = np.nan
            dic[str(year) + "_" + "O3_mean"] = np.nan
            dic[str(year) + "_" + "SO2_mean"] = np.nan
            dic[str(year) + "_" + "CO_mean"] = np.nan
            gas_year.append(dic)         
    gas_avg.append(gas_year)

def prepModel(gas):
    dates = pd.DataFrame()
    dates["Time"] = pd.date_range('1/1/2000', periods=192, freq='M')
    dates["Time"] = pd.to_datetime(dates["Time"], format='%Y-%m-%d').apply(lambda x: x.strftime('%Y-%m'))
    dates["Gas_Avg"] = getGasMonthlyAvg(gas)[:192]
    return dates

def evaluate_arima_model(gas, order):
    test_set = getGasMonthlyAvg(gas, month_year)[192:]
    model = ARIMA(prepModel(gas)["Gas_Avg"], order=order)
    model_fit = model.fit()
    pred = model_fit.forecast(steps=12)[0]
    error = mean_squared_error(test_set, pred)
    return error

def evaluate_models(gas, p_values, d_values, q_values):
    best_score, best_cfg = float("inf"), None
    for p in p_values:
        for d in d_values:
            for q in q_values:
                order = (p,d,q)
                try:
                    mse = evaluate_arima_model(gas, order)
                    if mse < best_score:
                        best_score, best_cfg = mse, order
                        print('ARIMA%s MSE=%.3f' % (order,mse))
                except:
                    continue
    print("Best ARIMA for gas {} is {} MSE={}".format(gas, best_cfg, best_score))

def varmaModel(gas):
    projections = []
    for frame in frames:
        df = frame.filter(regex=gas)
        df.loc[0] = years
        x = df.values.tolist()
        rand = [y+random() for y in x[0]]
        data = list(map(list, zip(rand, x[0])))
        model = VARMAX(data, order=(1, 1))
        model_fit = model.fit(disp=False)
        yhat = model_fit.forecast()
        projections.append(yhat)
    return projections

years = [2000,2001,2002,2003,2004,2005,2006,2007,2008,2009,2010,2011,2012,2013,2014,2015,2016,2017]
df_2000 = df_coast[["2000_CO_mean", "State"]]

data_bal = []

data_2000 = [dict(type='choropleth',
                colorscale = "RdBu",
                autocolorscale = False,
                locations = df_2000['State'],
                z = df_2000["2000_CO_mean"].astype(float),
                locationmode = 'USA-states',
                text = df_2000['State'],
                marker = dict(line = dict(color = 'rgb(255,255,255)',
                                          width = 2)),
                visible = True,
                colorbar = dict(title = "Parts per million"))]

data_bal.extend(data_2000)

for i in years[1:]:
    data_upd = [dict(type='choropleth',
                      colorscale = "RdBu",
                      autocolorscale = False,
                      locations = df_coast["State"],
                      z = df_coast[str(i) + "_" + "CO_mean"].astype(float),
                      locationmode = 'USA-states',
                      text = df_coast['State'],
                      marker = dict(line = dict(color = 'rgb(255,255,255)',
                                                width = 2)),
                      visible = True,
                      colorbar = dict(title = "Parts per million"))]
    
    data_bal.extend(data_upd)

steps = []
yr = 0
for i in range(0,len(data_bal)):
    step = dict(method = "restyle",
                args = ["visible", [False]*len(data_bal)],
                label = years[yr]) 
    step['args'][1][i] = True
    steps.append(step)
    yr += 1

sliders = [dict(active = 10,
                currentvalue = {"prefix": "Year: "},
                pad = {"t": 50},
                steps = steps)]    

layout = dict(title = 'CO averages on the Coast from 2000-2017',
              geo = dict(scope="usa",
                         projection=dict( type='albers usa' ),
                         showlakes = True,
                         lakecolor = 'rgb(255, 255, 255)'),
                         sliders = sliders)

fig = dict(data=data_bal, layout=layout)
iplot(fig, filename='CO pred')
